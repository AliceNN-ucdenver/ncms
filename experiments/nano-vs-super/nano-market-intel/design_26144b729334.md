## API Endpoint Specifications
| Method | Route | Description | Request TypeScript Interface | Response TypeScript Interface | Status Codes |
|--------|-------|-------------|------------------------------|-------------------------------|--------------|
| `POST` | `/api/auth/login` | Username + password → returns access & refresh JWTs | `{ username: string, password: string }` | `{ accessToken: string, refreshToken: string, expiresIn: number }` | `200` (success), `401` (invalid creds), `400` (validation error) |
| `POST` | `/api/auth/refresh` | Exchange valid refresh token for new pair | `{ refreshToken: string }` | `{ accessToken: string, refreshToken: string, expiresIn: number }` | `200`, `401` (invalid/expired refresh), `403` (revoked) |
| `POST` | `/api/auth/logout` | Revoke a refresh token (client must send token) | `{ refreshToken: string }` | `{ message: string }` | `200`, `400` (missing), `401` (already revoked) |
| `POST` | `/api/auth/mfa/verify` | Validate OTP sent via SMS/email | `{ email: string, otp: string }` | `{ success: boolean }` | `200`, `400` (bad OTP), `404` (email not found) |
| `GET` | `/api/me` | Authenticated user profile | **Headers**: `Authorization: Bearer <access>` | `{ id: string, email: string, role: Role }` | `200`, `401` (no valid token) |
| `GET` | `/api/health` | Liveness probe (no auth) | – | `{ status: 'ok' }` | `200` |

All routes are registered in `api/routes/*.routes.ts` and prefixed with `/api`.

## Authentication Middleware Implementation
### `src/auth/middleware/authMiddleware.ts`
```ts
import { Request, Response, NextFunction } from 'express';
import jwt, { JwtPayload } from 'jsonwebtoken';
import { config } from '../../config';
import { User } from '../models/User';
import redis from '../../utils/redisClient';

export async function verifyAccessToken(
  req: Request,
  _res: Response,
  next: NextFunction
) {
  const token = req.headers.authorization?.split(' ')[1];
  if (!token) return next(new Error('UNAUTHORIZED'));

  try {
    const payload = jwt.verify(token, config.JWT_PUBLIC_KEY) as JwtPayload;
    const user = await User.findOneBy({ id: payload.sub });
    if (!user) throw new Error('USER_NOT_FOUND');

    // Attach user to request for downstream handlers
    (req as any).user = user;
    next();
  } catch (err) {
    next(new Error('INVALID_TOKEN'));
  }
}
```

### Refresh Token Rotation (excerpt from `authMiddleware.ts`)
```ts
export async function rotateRefreshToken(
  req: Request,
  res: Response,
  next: NextFunction
) {
  const refresh = req.body.refreshToken;
  if (!refresh) return next();

  const tokenId = jwt.decode(refresh)?.jti;
  const stored = await User.findOne({
    where: { refreshTokenHash: await hashToken(refresh) },
  });

  if (!stored) return next(new Error('INVALID_REFRESH'));

  // Invalidate previous token in DB
  stored.refreshTokenHash = undefined;
  await stored.save();

  // Generate new pair and attach to response locals
  const { accessToken, newRefreshToken } = await generateTokens(stored.id, stored.role);
  res.locals.newAccess = accessToken;
  res.locals.newRefresh = newRefreshToken;
  next();
}
```

### 2. Input Validation (`src/auth/middleware/validateInput.ts`)
Uses **Zod** schemas derived from `token.schema.ts` and `user.schema.ts`.

```ts
import { z } from 'zod';
import { Request, Response, NextFunction } from 'express';

export const loginSchema = z.object({
  username: z.string().email(),
  password: z.string().min(8),
});

export const loginValidator = (req: Request, _res: Response, next: NextFunction) => {
  const result = loginSchema.safeParse(req.body);
  if (!result.success) {
    return next(new BadRequestError(result.error.format()));
  }
  next();
};
```

### 3. Token Rotation & Revocation (`src/auth/middleware/tokenRotation.ts`)
```ts
import crypto from 'crypto';
import redis from '../../utils/redisClient';
import { User } from '../models/User';

export async function storeRevokedJti(jti: string) {
  await redis.set(`revoked:${jti}`, '1', 'EX', 60_000); // keep for 60s
}

export async function isJtiRevoked(jti: string) {
  const exists = await redis.exists(`revoked:${jti}`);
  return !!exists;
}
```

### 4. CSRF Protection (stateless approach)
Since we use only JWTs in `Authorization` header, we rely on **SameSite=Lax** cookies for any state‑changing POST forms. If cookies are required, attach `csurf` middleware with a double‑submit token stored in a header (`X-CSRF-Token`).

```ts
import csurf from 'csurf';
import cookieParser from 'cookie-parser';

export const csrfProtection = csurf({
  cookie: true,
});
```

## Error Handling Patterns
### Custom Error Classes
```ts
export class AppError extends Error {
  public readonly statusCode: number;
  public readonly isOperational: boolean;
  constructor(message: string, statusCode = 500, isOperational = true) {
    super(message);
    this.statusCode = statusCode;
    this.isOperational = isOperational;
    Object.setPrototypeOf(this, new.target.prototype);
    Error.captureStackTrace(this, this.constructor);
  }
}

export class BadRequestError extends AppError {
  constructor(message: string) {
    super(message, 400);
  }
}
export class UnauthorizedError extends AppError {
  constructor(message: string) {
    super(message, 401);
  }
}
export class ForbiddenError extends AppError {
  constructor(message: string) {
    super(message, 403);
  }
}
export class NotFoundError extends AppError {
  constructor(message: string) {
    super(message, 404);
  }
}
```

### Global Error Handler (`src/middlewares/errorHandler.ts`)
```ts
import { AppError } from '../error';
import { config } from '../config';

export const errorHandler = (
  err: Error | AppError,
  _req: any,
  _res: any,
  // eslint-disable-next-line @typescript-eslint/no-undef
  _next: Function
) => {
  const statusCode = (err as AppError).statusCode || 500;
  const message = (err as any).message || 'Internal Server Error';

  // Log error (pseudocode)
  logger.error(err.stack);

  res.status(statusCode).json({
    error: {
      message,
      // In production hide stack trace
      ...(config.NODE_ENV === 'development' && { stack: (err as any).stack }),
    },
  });
};
```

When an error is thrown inside async route handlers, wrap with `asyncWrapper` (utility) to forward to the error handler.
